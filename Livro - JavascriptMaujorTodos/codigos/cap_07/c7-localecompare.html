<script type="text/javascript">
function slocalecompare(string2) {
	var string1 = document.form.string1.value;
	var string2 = document.form.string2.value;
	var resultado =  string1.localeCompare(string2);
		if(resultado <0) {
			document.form.resultado.value = 'A string "'+ string1 + '" precede a string "'+ string2 + '"';
		} else if (resultado > 0) {
			document.form.resultado.value = 'A string "'+ string1 + '" sucede a string "'+ string2 + '"';
		} else {		
			document.form.resultado.value = 'A string "'+ string1 + '" é igual à string "'+ string2 + '"';
		};
}
