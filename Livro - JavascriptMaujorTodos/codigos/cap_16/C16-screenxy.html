Eventos - Propriedades screenX e screenY

JavaScript

<script type="text/javascript">
var EventUtil = {
	getEvent: function(event) {
		return event ? event : window.event;
	}
}
window.onload = function() {	
	
	var demo = document.getElementById("demo");	
	var caixasTexto = document.getElementsByTagName("input");

	demo.onmousemove = function(event) {
	var event = EventUtil.getEvent(event);
	caixasTexto[0].value = event.screenX + " pixels";
	caixasTexto[1].value = event.screenY + " pixels";
	}
}
</script>
