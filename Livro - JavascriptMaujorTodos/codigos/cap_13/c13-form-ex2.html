<script type="text/javascript">
window.onload = function () {
var el, msg, mensagem, elBotao;
	el = document.getElementsByTagName("form")["form2"];
	msg = document.getElementById("msg");
	elBotao = document.getElementById("btn");	
	mensagem = "<h3>Atributos do primeiro elemento <code>form</code> do documento</h3>";
	mensagem += "<ul>";
	mensagem += "<li>name= " + el.name + "</li>";
	mensagem += "<li>action= " + el.action + "</li>";
	mensagem += "<li>method= " + el.method + "</li>";
	mensagem += "<li>target= " + el.target + "</li>";
	mensagem += "</ul>";	
	elBotao.onclick	= function() {
		msg.innerHTML = mensagem;
	}
}
