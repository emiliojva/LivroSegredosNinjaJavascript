Propriedades do objeto link

JavaScript

<script type="text/javascript">
window.onload = function () {
var el, msg, mensagem, elBotao;
	el = document.getElementsByTagName("link")
	msg = document.getElementById("msg")
	elBotao = document.getElementById("btn");	
	mensagem = "<h3>Atributos do primeiro elemento <code>link</code> do documento</h3>";
	mensagem += "<ul>";
	mensagem += "<li>href= " + el[0].href + "</li>";
	mensagem += "<li>rel= " + el[0].rel + "</li>";
	mensagem += "<li>type= " + el[0].type + "</li>";
	mensagem += "<li>media= " + el[0].media + "</li>";
	mensagem += "</ul>";	
	elBotao.onclick	= function() {
		msg.innerHTML = mensagem;
	el[0].disabled = true;
	}
}
<script>
