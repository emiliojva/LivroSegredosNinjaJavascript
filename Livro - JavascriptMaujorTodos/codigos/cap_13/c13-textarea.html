Propriedades do objeto textarea

JavaScript

<script type="text/javascript">
window.onload = function () {
var el, msg, elBotao, mensagem; 
	el = document.getElementById("coment");
	msg = document.getElementById("msg");
	elBotao = document.getElementById("btn");	
	mensagem = "<h3>Propriedades do elemento <code>textarea</code></h3>";
	mensagem += "<ul>";
	mensagem += "<li>nome do form: " + el.form.name + "</li>";
	mensagem += "<li>atributo cols: " + el.cols + "</li>";
	mensagem += "<li>atributo rows: " + el.rows + "</li>";
	mensagem += "<li>tipo de controle: " + el.type + "</li>";
	mensagem += "<li>defaultValue: " + el.defaultValue + "</li>";
	mensagem += "</ul>";
	
	elBotao.onclick	= function() {
		msg.innerHTML = mensagem;
		el.value = "Conteúdo inserido com a propriedade value"
	}

}
<script>
