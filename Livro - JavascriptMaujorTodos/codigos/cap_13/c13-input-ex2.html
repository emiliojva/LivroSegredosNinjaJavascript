Métodos do objeto input - Exemplo 2

JavaScript

<script type="text/javascript">
window.onload = function () {
var el, foco, seleciona; 
	el = document.getElementsByTagName("input");
	foco = document.getElementById("foco");
	seleciona = document.getElementById("seleciona");
		foco.onclick = function() {
			el["nome"].focus();
		}
		
		seleciona.onclick = function() {
			el["data"].select();
		}
}
<script>
