Métodos - setInterval() e clearInterval()

JavaScript

<script type="text/javascript">
var i = 0;
var contador=null;
var contando = false;	
	
	function contar() {
		i++;
		document.forms[0].elements[0].value = i;
	}
	
	function iniciarContador() {	
		if(!contando) {
		contando = true;
		contador = setInterval(contar, 1000);
		}
	}
	
	function pararContador() {	
		clearInterval(contador);	
		i=0;
		contando = false;
	}
<script>
